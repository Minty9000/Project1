// Create Users table. Includes roles to distinguish between customers, admins, and customer reps

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role VARCHAR(50) NOT NULL
);

// Creates Trainlines table. Origin and Destination reference the Stations table, so the Stations table needs to be created first

CREATE TABLE Trainlines (
    tlid INT PRIMARY KEY auto_increment,
    name VARCHAR(100) NOT NULL,
    origin int,
    destination int,
    travelTime int,
    numOfStops int,
    fare float,
    FOREIGN KEY (origin) REFERENCES stations (sid),
    FOREIGN KEY (destination) REFERENCES stations (sid)
);

// Creates Stops table. This keeps track of the stops and their associated train lines. The Trainlines and Stations tables need to be
// created first.

CREATE TABLE Stops (
    tlid INT,
    sid INT,
    travelTime int,
    stopNumber int,
    PRIMARY KEY (tlid, sid),
    FOREIGN KEY (tlid) REFERENCES trainlines (tlid),
    FOREIGN KEY (sid) REFERENCES stations (sid)
);
